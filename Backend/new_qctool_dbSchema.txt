1. CREATE DATABASE "QC_TOOL";


2. CREATE SCHEMA IF NOT EXISTS qc_portal;
SET search_path TO qc_portal;


3. CREATE EXTENSION IF NOT EXISTS "uuid-ossp";


4. CREATE TYPE user_role AS ENUM (
    'SuperAdmin',
    'QC_Supervisor',
    'QC_User',
    'Upload_Supervisor',
    'Vendor',
    'Scanning_Operator'
);


CREATE TYPE upload_status AS ENUM ('Pending','In_Progress','Completed','Failed');


CREATE TYPE conversion_status AS ENUM (
    'Tiff_Received','Jpeg_Converting','Jpeg_Converted','QC_Moved','Failed'
);


CREATE TYPE file_type AS ENUM ('TIFF','JPEG','PNG');


CREATE TYPE qc_batch_status AS ENUM (
    'Allocated','QC_Pending','QC_In_Progress','Completed'
);


CREATE TYPE qc_status AS ENUM (
    'Pending','Approved','Rejected','Needs_Rework'
);


5. CREATE TABLE users (
    user_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name VARCHAR(255) NOT NULL,
    username VARCHAR(100) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    user_role user_role NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_by UUID REFERENCES users(user_id) ON DELETE SET NULL,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT username_length CHECK (length(username) >= 3)
);


CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_role ON users(user_role);
CREATE INDEX idx_users_created_by ON users(created_by);


6. CREATE TABLE projects (
    project_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_code VARCHAR(50) UNIQUE NOT NULL,
    project_name VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT project_code_format CHECK (project_code ~ '^[A-Z0-9_]+$')
);


CREATE INDEX idx_projects_code ON projects(project_code);
CREATE INDEX idx_projects_created_by ON projects(created_by);




7. CREATE TABLE source (
    source_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_id UUID NOT NULL REFERENCES projects(project_id),
    source_code VARCHAR(50) NOT NULL,
    source_name VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT source_code_format CHECK (source_code ~ '^[A-Z0-9_]+$'),
    UNIQUE (project_id, source_code)
);


CREATE INDEX idx_source_project ON source(project_id);
CREATE INDEX idx_source_code ON source(source_code);


8. CREATE TABLE location (
    location_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_id UUID NOT NULL REFERENCES projects(project_id),
    source_id UUID NOT NULL REFERENCES source(source_id),
    location_code VARCHAR(50) NOT NULL,
    location_name VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT location_code_format CHECK (location_code ~ '^[A-Z0-9_]+$'),
    UNIQUE (source_id, location_code)
);


CREATE INDEX idx_location_source ON location(source_id);
CREATE INDEX idx_location_project ON location(project_id);




9. CREATE TABLE record_owners (
    record_owner_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_id UUID NOT NULL REFERENCES projects(project_id),
    source_id UUID NOT NULL REFERENCES source(source_id),
    location_id UUID NOT NULL REFERENCES location(location_id),
    record_owner_code VARCHAR(50) NOT NULL,
    record_owner_name VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT record_owner_code_format CHECK (record_owner_code ~ '^[A-Z0-9_]+$'),
    UNIQUE (location_id, record_owner_code)
);


CREATE INDEX idx_record_owner_location ON record_owners(location_id);
CREATE INDEX idx_record_owner_project ON record_owners(project_id);






10. CREATE SEQUENCE record_code_seq START 1 INCREMENT 1;




11. CREATE TABLE record_name (
    record_name_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_id UUID NOT NULL REFERENCES projects(project_id),
    record_code VARCHAR(7) UNIQUE NOT NULL
        DEFAULT ('B' || LPAD(nextval('record_code_seq')::text, 6, '0')),
    record_name VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT record_code_format CHECK (record_code ~ '^B[0-9]{6}$')
);


CREATE INDEX idx_record_name_code ON record_name(record_code);
CREATE INDEX idx_record_name_project ON record_name(project_id);


12. CREATE TABLE record_type (
    record_type_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    record_type_code VARCHAR(50) NOT NULL,
    record_type_name VARCHAR(255) NOT NULL,
    source_id UUID NOT NULL REFERENCES source(source_id),
    created_by UUID NOT NULL REFERENCES users(user_id),
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (source_id, record_type_code)
);


CREATE INDEX idx_record_type_source ON record_type(source_id);




13. CREATE TABLE vendor_allocation (
    vendor_allocation_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    project_id UUID NOT NULL REFERENCES projects(project_id),
    source_id UUID NOT NULL REFERENCES source(source_id),
    location_id UUID NOT NULL REFERENCES location(location_id),
    record_owner_id UUID NOT NULL REFERENCES record_owners(record_owner_id),
    allocated_to_vendor UUID NOT NULL REFERENCES users(user_id),
    allocated_by_supervisor UUID NOT NULL REFERENCES users(user_id),
    is_active BOOLEAN DEFAULT TRUE,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (project_id, source_id, location_id, record_owner_id, allocated_to_vendor)
);


CREATE INDEX idx_vendor_project ON vendor_allocation(project_id);
CREATE INDEX idx_vendor_vendor ON vendor_allocation(allocated_to_vendor);




14. CREATE TABLE scanning_operator_allocation (
    scanning_operator_allocation_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vendor_allocation_id UUID NOT NULL REFERENCES vendor_allocation(vendor_allocation_id),
    allocated_to_operator UUID NOT NULL REFERENCES users(user_id),
    is_active BOOLEAN DEFAULT TRUE,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (vendor_allocation_id, allocated_to_operator)
);


CREATE INDEX idx_scan_operator ON scanning_operator_allocation(allocated_to_operator);




15. CREATE TABLE batch (
    batch_uid UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    batch_id VARCHAR(255) UNIQUE NOT NULL,
    scanning_operator_allocation_id UUID NOT NULL REFERENCES scanning_operator_allocation(scanning_operator_allocation_id),
    source_id UUID NOT NULL REFERENCES source(source_id),
    location_id UUID NOT NULL REFERENCES location(location_id),
    record_owner_id UUID NOT NULL REFERENCES record_owners(record_owner_id),
    record_name_id UUID NOT NULL REFERENCES record_name(record_name_id),
    record_type_id UUID NOT NULL REFERENCES record_type(record_type_id),
    total_count INTEGER NOT NULL CHECK (total_count > 0),
    upload_count INTEGER DEFAULT 0 CHECK (upload_count >= 0),
    is_complete BOOLEAN DEFAULT FALSE,
    is_partial BOOLEAN DEFAULT FALSE,
    is_reupload BOOLEAN DEFAULT FALSE,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT batch_type_exclusive CHECK (
        (is_complete::int + is_partial::int + is_reupload::int) = 1
    )
);


CREATE INDEX idx_batch_batch_id ON batch(batch_id);
CREATE INDEX idx_batch_operator ON batch(scanning_operator_allocation_id);






16. CREATE TABLE upload (
    upload_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    batch_uid UUID NOT NULL REFERENCES batch(batch_uid),
    completed_count INTEGER DEFAULT 0,
    s3_folder_path VARCHAR(500) NOT NULL,
    upload_status upload_status DEFAULT 'Pending',
    uploaded_by UUID NOT NULL REFERENCES users(user_id),
    upload_start_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    upload_end_date TIMESTAMP,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX idx_upload_batch ON upload(batch_uid);
CREATE INDEX idx_upload_status ON upload(upload_status);


17. CREATE TABLE image (
    image_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    upload_id UUID NOT NULL REFERENCES upload(upload_id),
    batch_uid UUID NOT NULL REFERENCES batch(batch_uid),
    image_name VARCHAR(500) NOT NULL,
    original_s3_path VARCHAR(1000) NOT NULL,
    qc_s3_path VARCHAR(1000),
    original_file_type file_type NOT NULL,
    converted_file_type file_type,
    conversion_status conversion_status DEFAULT 'Tiff_Received',
    file_size_bytes BIGINT CHECK (file_size_bytes > 0),
    upload_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (batch_uid, image_name)
);


CREATE INDEX idx_image_batch ON image(batch_uid);
CREATE INDEX idx_image_status ON image(conversion_status);


18. CREATE TABLE qc_allocation (
    qc_allocation_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    batch_uid UUID NOT NULL REFERENCES batch(batch_uid),
    allocated_to_qc_user UUID NOT NULL REFERENCES users(user_id),
    allocated_by_supervisor UUID NOT NULL REFERENCES users(user_id),
    allocation_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    qc_batch_status qc_batch_status DEFAULT 'Allocated',
    qc_completed_date TIMESTAMP,
    UNIQUE (batch_uid, allocated_to_qc_user)
);


CREATE INDEX idx_qc_alloc_batch ON qc_allocation(batch_uid);
CREATE INDEX idx_qc_alloc_user ON qc_allocation(allocated_to_qc_user);






19. CREATE TABLE qc (
    qc_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    qc_allocation_id UUID NOT NULL REFERENCES qc_allocation(qc_allocation_id),
    image_id UUID NOT NULL REFERENCES image(image_id),
    qc_status qc_status DEFAULT 'Pending',
    orientation_error BOOLEAN DEFAULT FALSE,
    remarks TEXT,
    qc_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (qc_allocation_id, image_id)
);


CREATE INDEX idx_qc_status ON qc(qc_status);
CREATE INDEX idx_qc_image ON qc(image_id);
